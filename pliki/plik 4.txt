#include <stdlib.h>

int tablica (int dl)
{
int i;
int  *p=malloc(dl);
for (i=0; i<dl; i++)
    p [i]=2*i+1;
if (!p)
    {
        perror("malloc");
        return EXIT_FAILURE;
    }
return p;
}
